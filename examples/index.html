这是模块的输出文件，请 require 或 seajs.use 以上 ID 来使用本文件，具体 API 请阅读文档。

Tabs - 普通标签页

<div id="demo1">
  <ul class="tab-nav">
    <li class="tab-hd">item</li>
    <li class="tab-hd">item</li>
    <li class="tab-hd">item</li>
  </ul>
  <div class="tab-strip"></div>
  <div class="tab-bd">
    <div class="bd">a</div>
    <div class="bd">b</div>
    <div class="bd">c</div>
  </div>
</div>

<script type="text/javascript">
seajs.use(['kjui/tab/0.0.1/tab'], function(Tab) {
  tabs = new Tab({
    element: '#demo1',
    triggers: '.tab-nav li',
    panels: '.tab-bd div'
  });
});
</script>

auto-render

<div id="demo2" data-widget="tab">
  <ul class="tab-nav" data-role="nav">
    <li class="tab-hd">item</li>
    <li class="tab-hd">item</li>
    <li class="tab-hd">item</li>
  </ul>
  <div class="tab-strip"></div>
  <div class="tab-bd" data-role="content">
    <div class="bd">a</div>
    <div class="bd">b</div>
    <div class="bd">c</div>
  </div>
</div>

<script type="text/javascript">
seajs.use(['widget'], function(Widget) {
  Widget.autoRenderAll();
});
</script>
